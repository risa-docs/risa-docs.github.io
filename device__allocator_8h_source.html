45 template <class T>
46 class device_allocator<T, memory_layout::pointer_1D>
47 {
48 public:
49 static constexpr auto mem_layout = memory_layout::pointer_1D;
50 static constexpr auto mem_location = memory_location::device;
51 static constexpr auto alloc_needs_pitch = false;
52
53 using value_type = T;
54 using pointer = value_type*;
55 using const_pointer = const pointer;
56 using size_type = std::size_t;
57 using difference_type = std::ptrdiff_t;
58 using propagate_on_container_copy_assignment = std::true_type;
59 using propagate_on_container_move_assignment = std::true_type;
60 using propagate_on_container_swap = std::true_type;
61 using is_always_equal = std::true_type;
62
63 template <class Deleter>
64 using smart_pointer = unique_ptr<T, Deleter, alloc_needs_pitch, mem_location, false>;
65
66 template <class U>
67 struct rebind
68 {
69 using other = device_allocator<U, mem_layout>;
70 };
71
72 device_allocator() noexcept = default;
73 device_allocator(const device_allocator& other) noexcept = default;
74
75 template <class U, memory_layout uml>
76 device_allocator(const device_allocator<U, uml>&) noexcept
77 {
78 static_assert(std::is_same<T, U>::value && mem_layout == uml, "Attempting to copy incompatible device allocator");
79 }
80
81 ~device_allocator() = default;
82
83 auto allocate(size_type n) -> pointer
84 {
85 auto ptr = static_cast<pointer>(nullptr);
86 auto err = cudaMalloc(reinterpret_cast<void**>(&ptr), n * sizeof(value_type));
87 if(err != cudaSuccess)
88 detail::throw_error(err);
89 return pointer{ptr};
90 }
91
92 auto deallocate(pointer p, size_type = 0) noexcept -> void
93 {
94 auto err = cudaFree(reinterpret_cast<void*>(p));
95 if(err != cudaSuccess)
96 std::exit(err);
97 }
98 };

197 auto pitched_ptr = cudaPitchedPtr{};
