37template <class T, class CopyPolicy, class Ptr>
38class base_ptr : public CopyPolicy
39{
40 public:
41 using pointer = typename Ptr::pointer;
42 using element_type = typename Ptr::element_type;
43 using deleter_type = typename Ptr::deleter_type;
44 using underlying_type = Ptr;
45
46 public:
47 constexpr base_ptr() noexcept : ptr_{} {}
48
49 constexpr base_ptr(std::nullptr_t p) noexcept : ptr_{p} {}
50
51 base_ptr(Ptr ptr) noexcept : ptr_{std::move(ptr)} {}
52
53 base_ptr(base_ptr&& other) noexcept : ptr_{std::move(other.ptr_)} {}
54
55 ~base_ptr() = default;
56
57 inline auto operator=(base_ptr&& r) noexcept -> base_ptr&
58 {
59 ptr_ = std::move(r.ptr_);
60 return *this;
61 }
62
63 inline auto release() noexcept -> pointer { return ptr_.release(); }
64 inline auto reset(pointer ptr = pointer()) noexcept -> void { ptr_.reset(ptr); }
65 template <class U>
66 auto reset(U) noexcept -> void = delete;
67 inline auto reset(std::nullptr_t p) noexcept -> void { ptr_.reset(p); }
68 inline auto swap(base_ptr& other) noexcept -> void { ptr_.swap(other.ptr_); }
69 inline auto get() const noexcept -> pointer { return ptr_.get(); }
70 inline auto get_deleter() noexcept -> deleter_type { return ptr_.get_deleter(); }
71 inline auto get_deleter() const noexcept -> const deleter_type { return ptr_.get_deleter(); }
72 explicit inline operator bool() const noexcept { return ptr_.operator bool(); }
73 inline auto operator[](std::size_t i) const -> element_type& { return ptr_[i]; }
74
75 // this is nasty but currently I see no other way to access the ptr_ of different instances
76 inline const Ptr& get_underlying() const noexcept { return ptr_; }
77
78 protected:
79 template <class Dest, class Src, class... Args>
80 inline auto copy(Dest& dest, const Src& src, Args... args) -> void
81 {
82 CopyPolicy::copy(dest, src, args...);
83 }
84
85 private:
86 base_ptr(const base_ptr&) = delete;
87 auto operator=(const base_ptr&) -> base_ptr& = delete;
88
89 protected:
90 Ptr ptr_;
91};

129template <class T, class CopyPolicy, class Ptr>
130class ptr : public base_ptr<T, CopyPolicy, Ptr>
131{
132 private:
133 using base = base_ptr<T, CopyPolicy, Ptr>;
134 std::size_t size_;
135
136 public:
137 using pointer = typename base::pointer;
138 using element_type = typename base::element_type;
139 using deleter_type = typename base::deleter_type;
140 using underlying_type = typename base::underlying_type;
141 static constexpr auto has_pitch = false;
142
143 public:
144 constexpr ptr() noexcept : base(), size_{0u} {}
145
146 constexpr ptr(std::nullptr_t p) noexcept : base(p), size_{0u} {}
147
148 ptr(Ptr p, std::size_t s) noexcept : base(std::move(p)), size_{s} {}
149
150 ptr(ptr&& other) noexcept : base(std::move(other)), size_{other.size_} {}
151
152 ~ptr() = default;
153
154 inline auto operator=(ptr&& r) noexcept -> ptr&
155 {
156 size_ = r.size_;
157 base::operator=(std::move(r));
158 return *this;
159 }
160
161 inline auto size() const noexcept -> std::size_t { return size_; }
162
163 private:
164 ptr(const ptr&) = delete;
165 auto operator=(const ptr&) -> ptr& = delete;
166};
