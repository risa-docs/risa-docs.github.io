71 auto requestMemory(unsigned int idx) -> type
72 {
73 // std::lock_guard<std::mutex> lock(memoryManagerMutex_);
74 auto lock = std::unique_lock<std::mutex>{memoryManagerMutex_};
75 if(memoryPool_.size() <= idx)
76 throw std::runtime_error("cuda::MemoryPool: Stage needs to be registered first.");
77 while(memoryPool_[idx].empty()) //if all memory for the stage with this idx is given out already
78 {
79 cv_.wait(lock);//wait until some becomes availible
80 }
81 auto ret = std::move(memoryPool_[idx].back());
82 memoryPool_[idx].pop_back();
83 return ret;
84 }

95 auto returnMemory(type&& img) -> void
96 {
97 if(memoryPool_.size() <= img.memoryPoolIndex())
98 throw std::runtime_error("cuda::MemoryPool: Stage needs to be registered first.");
99 std::lock_guard<std::mutex> lock{memoryManagerMutex_};
100 memoryPool_[img.memoryPoolIndex()].push_back(std::move(img));
101 cv_.notify_one();//new memory availible, wake up a starved requestMemory-caller(if any)
102 }

116 auto registerStage(const int& numberOfElements, const size_t& size) -> int
117 {
118 // lock, to ensure thread safety
119 std::lock_guard<std::mutex> lock(memoryManagerMutex_);
120 std::vector<type> memory;
121 int index = memoryPool_.size();//get next unused index to ascertain that stage-inexes are given in ascending order (for size 3, {0,1,2} are taken-> use 3 next)
122 for(int i = 0; i < numberOfElements; i++)
123 {
124 auto ptr = MemoryManager::make_ptr(size);
125 auto img = type{size, 0, 0, std::move(ptr)};//make one of numberOfElements images
126 img.setMemPoolIdx(index);//tell the image the index of the stage it belongs to
127 memory.push_back(std::move(img));
128 }
129 memoryPool_.push_back(std::move(memory));//memory is now a vector of images, of length numberOfElements, enqueue it. This means that the stage with index "index" now has reserved "memory".
130 return index;
131 }
132
134
139 auto freeMemory(const unsigned int idx) -> void
140 {
141 for(auto& ele : memoryPool_[idx]) //for each img that this stage reserved
142 {
143 ele.invalid(); //invalidate image
144 }
145 memoryPool_[idx].clear(); //clean up Vector<img> , that held the memory for this stage
146 }
