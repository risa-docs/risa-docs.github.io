68  /* FIXME: In C++17 the constructor will be templated to template<class U>(U ptr). Change the
69  * following constructor accordingly as soon as CUDA supports C++17
70  */
71  explicit unique_ptr(pointer ptr) noexcept
72  : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_{}
73  {}

75  /*
76  * The following 2 constructors construct a unique_ptr object as follows:
77  * a) If Deleter is non-reference type A, then the signatures are:
78  * unique_ptr(pointer p, const A& d1);
79  * unique_ptr(pointer p, A&& d2);
80  * b) If Deleter is an lvalue-reference type A&, then the signatures are:
81  * unique_ptr(pointer p, A& d1);
82  * unique_ptr(pointer p, A&& d2);
83  * c) If Deleter is an lvalue-reference type const A&, then the signatures are:
84  * unique_ptr(pointer p, const A& d1);
85  * unique_ptr(pointer p, const A&& d2);
86  */
87  unique_ptr(pointer ptr,
88  typename std::conditional<std::is_reference<deleter_type>::value, deleter_type, const deleter_type&>::type d1) noexcept
89  : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_(d1)
90  {}
91 
92  unique_ptr(pointer ptr,
93  typename std::remove_reference<deleter_type>::type&& d2) noexcept
94  : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_(d2)
95  {}
96 
97  unique_ptr(unique_ptr&& u)
98  {
99  if(u != nullptr)
100  {
101  ptr_ = u.ptr_;
102  pitch_ = u.pitch_;
103  u.ptr_ = nullptr;
104  u.pitch_ = 0;
105 
106  if(std::is_reference<deleter_type>::value)
107  deleter_ = u.deleter_;
108  else
109  deleter_ = std::move(u.deleter_);
110  }
111  else
112  {
113  ptr_ = nullptr;
114  pitch_ = 0;
115  }
116  }
117 
118  ~unique_ptr()
119  {
120  if(ptr_ != nullptr)
121  deleter_(ptr_);
122  }

128 
129  if(r != nullptr)

131  ptr_ = r.ptr_;
132  pitch_ = r.pitch_;
133  r.ptr_ = nullptr;
134  r.pitch_ = 0;
135 
136  if(std::is_reference<deleter_type>::value)
137  deleter_ = r.deleter_;
138  else

415  template <class T>

417  {

425  template <class T>
426  auto make_unique_device(std::size_t x, std::size_t y) -> pitched_device_ptr<T>
427  {
428  auto ptr = static_cast<T*>(nullptr);
429  auto pitch = std::size_t{};
430  auto err = cudaMallocPitch(reinterpret_cast<void**>(&ptr), &pitch, x * sizeof(T), y);
431  if(err != cudaSuccess)
432  detail::throw_error(err);
433  return pitched_device_ptr<T>{pitched_ptr<T>{reinterpret_cast<T*>(ptr), pitch}};
434  }
