49 template <class T, class Deleter, memory_location loc>
50 class unique_ptr<T, Deleter, true, loc, false> // As of now (CUDA 8.0) pinned host memory is never pitched
51 {
52 public:
53 using pointer = pitched_ptr<T>;
54 using element_type = T;
55 using deleter_type = Deleter;
56
57 static constexpr auto mem_location = loc;
58 static constexpr auto pitched_memory = true;
59 static constexpr auto pinned_memory = false;
60
61 constexpr unique_ptr() noexcept : ptr_{nullptr}, pitch_{0u}, deleter_{}
62 {}
63
64 constexpr unique_ptr(std::nullptr_t) noexcept : ptr_{nullptr}, pitch_{0u}, deleter_{}
65 {}
66
67 /* FIXME: In C++17 the constructor will be templated to template<class U>(U ptr). Change the
68 * following constructor accordingly as soon as CUDA supports C++17
69 */
70 explicit unique_ptr(pointer ptr) noexcept
71 : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_{}
72 {}
73
74 /*
75 * The following 2 constructors construct a unique_ptr object as follows:
76 * a) If Deleter is non-reference type A, then the signatures are:
77 * unique_ptr(pointer p, const A& d1);
78 * unique_ptr(pointer p, A&& d2);
79 * b) If Deleter is an lvalue-reference type A&, then the signatures are:
80 * unique_ptr(pointer p, A& d1);
81 * unique_ptr(pointer p, A&& d2);
82 * c) If Deleter is an lvalue-reference type const A&, then the signatures are:
83 * unique_ptr(pointer p, const A& d1);
84 * unique_ptr(pointer p, const A&& d2);
85 */
86 unique_ptr(pointer ptr,
87 typename std::conditional<std::is_reference<deleter_type>::value, deleter_type, const deleter_type&>::type d1) noexcept
88 : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_(d1)
89 {}
90
91 unique_ptr(pointer ptr,
92 typename std::remove_reference<deleter_type>::type&& d2) noexcept
93 : ptr_{ptr.ptr()}, pitch_{ptr.pitch()}, deleter_(d2)
94 {}
95
96 unique_ptr(unique_ptr&& u)
97 {
98 if(u != nullptr)
99 {
100 ptr_ = u.ptr_;
101 pitch_ = u.pitch_;
102 u.ptr_ = nullptr;
103 u.pitch_ = 0;
104
105 if(std::is_reference<deleter_type>::value)
106 deleter_ = u.deleter_;
107 else
108 deleter_ = std::move(u.deleter_);
109 }
110 else
111 {
112 ptr_ = nullptr;
113 pitch_ = 0;
114 }
115 }
116
117 ~unique_ptr()
118 {
119 if(ptr_ != nullptr)
120 deleter_(ptr_);
121 }
122
123 auto operator=(unique_ptr&& r) noexcept -> unique_ptr&
124 {
125 if(ptr_ != nullptr)
126 deleter_(ptr_);
127
128 if(r != nullptr)
129 {
130 ptr_ = r.ptr_;
131 pitch_ = r.pitch_;
132 r.ptr_ = nullptr;
133 r.pitch_ = 0;
134
135 if(std::is_reference<deleter_type>::value)
136 deleter_ = r.deleter_;
137 else
138 deleter_ = std::move(r.deleter_);
139 }
140 else
141 {
142 ptr_ = nullptr;
143 pitch_ = 0;
144 }
145
146 return *this;
147 }
148
149 auto operator=(std::nullptr_t) noexcept -> unique_ptr&
150 {
151 if(ptr_ != nullptr)
152 deleter_(ptr_);
153
154 ptr_ = nullptr;
155 pitch_ = 0;
156 return *this;
157 }
158
159 auto release() noexcept -> pointer
160 {
161 auto ret = pointer{ptr_, pitch_};
162 ptr_ = nullptr;
163 return ret;
164 }
165
166 /*
167 * FIXME: Remove this method as soon as CUDA supports C++17
168 */
169 auto reset(pointer ptr = pointer()) noexcept -> void
170 {
171 auto old_ptr = ptr_;
172 ptr_ = ptr.ptr();
173 pitch_ = ptr.pitch();
174
175 if(old_ptr != nullptr)
176 deleter_(old_ptr);
177 }
178
179 /* FIXME: Change this to standard behaviour as soon as CUDA supports C++17. */
180 template <class U>
181 auto reset(U) noexcept -> void = delete;
182
183 auto reset(std::nullptr_t) noexcept -> void
184 {
185 auto old_ptr = ptr_;
186 ptr_ = nullptr;
187 pitch_ = 0;
188
189 if(old_ptr != nullptr)
190 deleter_(old_ptr);
191 }
192
193 auto swap(unique_ptr& other) noexcept -> void
194 {
195 std::swap(ptr_, other.ptr_);
196 std::swap(pitch_, other.pitch_);
197 std::swap(deleter_, other.deleter_);
198 }
199
200 auto get() const noexcept -> element_type*
201 {
202 return ptr_;
203 }
204
205 auto get_deleter() noexcept -> deleter_type&
206 {
207 return deleter_;
208 }
209
210 auto get_deleter() const noexcept -> const deleter_type&
211 {
212 return deleter_;
213 }
214
215 explicit operator bool() const noexcept
216 {
217 return ptr_ != nullptr;
218 }
219
220 auto pitch() const noexcept -> std::size_t
221 {
222 return pitch_;
223 }
224
225 private:
226 element_type* ptr_;
227 std::size_t pitch_;
228 deleter_type deleter_;
229 };

414 template <class T>
415 auto make_unique_device(std::size_t n) -> device_ptr<T>
416 {
417 auto ptr = static_cast<T*>(nullptr);
418 auto err = cudaMalloc(reinterpret_cast<void**>(&ptr), n * sizeof(T));
419 if(err != cudaSuccess)
420 detail::throw_error(err);
421 return device_ptr<T>{ptr};
422 }

424 template <class T>
425 auto make_unique_device(std::size_t x, std::size_t y) -> pitched_device_ptr<T>
426 {
427 auto ptr = static_cast<T*>(nullptr);
428 auto pitch = std::size_t{};
429 auto err = cudaMallocPitch(reinterpret_cast<void**>(&ptr), &pitch, x * sizeof(T), y);
430 if(err != cudaSuccess)
431 detail::throw_error(err);
432 return pitched_device_ptr<T>{pitched_ptr<T>{reinterpret_cast<T*>(ptr), pitch}};
433 }

435 template <class T>
436 auto make_unique_device(std::size_t x, std::size_t y, std::size_t z) -> pitched_device_ptr<T>
437 {
438 auto extent = make_cudaExtent(x * sizeof(T), y, z);
439 auto cuda_pitched_ptr = cudaPitchedPtr{};

444 }

494 template <class T1, class D1, bool p1, memory_location l1, bool pn1, class T2, class D2, bool p2, memory_location l2, bool pn2>
495 auto operator==(const unique_ptr<T1, D1, p1, l1, pn2>& x, const unique_ptr<T2, D2, p2, l2, pn2>& y) -> bool
496 {
497 if(l1 != l2) return false; // same location?
498 if(p1 != p2) return false; // same pointer type?
499 if(pn1 != pn2) return false; // are both pinned / not pinned?
500 return x.get() == y.get();
501 }
