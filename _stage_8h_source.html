73  auto run() -> void
74  {
75  std::vector<std::thread> push_threads;
76  std::vector<std::thread> take_threads;
77  for(int i = 0; i < numberOfInputs_; i++)
78  {
79  push_threads.push_back(std::thread{&Stage::push, this, i});
80  }
81  for(int i = 0; i < numberOfOutputs_; i++)
82  {
83  take_threads.push_back(std::thread{&Stage::take, this, i});
84  }
85 
86  for(int i = 0; i < numberOfInputs_; i++)
87  {
88  push_threads[i].join();
89  };
90  for(int i = 0; i < numberOfOutputs_; i++)
91  {
92  take_threads[i].join();
93  };
94  }

96  auto push(int inputIdx = 0) -> void
97  {
98  while(true)
99  {
100  auto img = this->input_queues_[inputIdx].take();
101  if(img.valid())
102  Implementation::process(std::move(img), inputIdx);
103  else
104  {
105  // received poisonous pill, time to die
106  Implementation::process(std::move(img), inputIdx);
107  break;
108  }
109  }
110  }

112  auto take(int outputIdx = 0) -> void
113  {

123  this->output(std::move(result), outputIdx);
124  break;
125  }
126  }
127  }
