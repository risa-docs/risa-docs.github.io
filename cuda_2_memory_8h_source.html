81template <class T, class Deleter, Target t>
82class unique_ptr
83{
84 public:
85 using pointer = T*;
86 using element_type = T;
87 using deleter_type = Deleter;
88 static constexpr auto target = Target{t};
89
90 public:
91 constexpr unique_ptr() noexcept : ptr_{} {}
92
93 constexpr unique_ptr(std::nullptr_t) noexcept : ptr_{nullptr} {}
94
95 unique_ptr(pointer p) noexcept : ptr_{p} {}
96
97 unique_ptr(unique_ptr&& other) noexcept : ptr_{std::move(other.ptr_)} {}
98
99 inline auto operator=(unique_ptr&& r) noexcept -> unique_ptr&
100 {
101 ptr_ = std::move(r.ptr_);
102 return *this;
103 }
104
105 inline auto operator=(std::nullptr_t) noexcept -> unique_ptr&
106 {
107 ptr_ = nullptr;
108 return *this;
109 }
110
111 inline auto release() noexcept -> pointer { return ptr_.release(); }
112 inline auto reset(pointer ptr = pointer()) noexcept -> void { ptr_.reset(ptr); }
113
114 template <class U>
115 auto reset(U) noexcept -> void = delete;
116
117 inline auto reset(std::nullptr_t) noexcept -> void { ptr_.reset(); }
118 inline auto swap(unique_ptr& other) noexcept -> void { ptr_.swap(other.ptr_); }
119 inline auto get() const noexcept -> pointer { return ptr_.get(); }
120 inline auto get_deleter() noexcept -> Deleter& { return ptr_.get_deleter(); }
121 inline auto get_deleter() const noexcept -> const Deleter& { return ptr_.get_deleter(); }
122 explicit inline operator bool() const noexcept { return bool(ptr_); }
123 inline auto operator[](std::size_t i) const -> T& { return ptr_[i]; }
124
125 private:
126 unique_ptr(const unique_ptr&) = delete;
127 auto operator=(const unique_ptr&) -> unique_ptr& = delete;
128
129 private:
130 std::unique_ptr<T[], Deleter> ptr_;
131};

174class sync_copy_policy
175{
176 protected:
177 ~sync_copy_policy() = default;
178
179 /* 1D copies*/
180 template <class Dest, class Src>
181 inline auto copy(Dest& dest, const Src& src, std::size_t size) const -> void
182 {
183 CHECK(cudaMemcpy(dest.get(), src.get(), size,
184 detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value));
185 }
186
187 /* 2D copies */
188 template <class Dest, class Src>
189 inline auto copy(Dest& dest, const Src& src, std::size_t width, std::size_t height) const -> void
190 {
191 CHECK(cudaMemcpy2D(dest.get(), dest.pitch(), src.get(), src.pitch(),
192 width * sizeof(typename Src::element_type), height,
193 detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value));
194 }
195
196 /* 3D copies */
197 template <class Dest, class Src>
198 inline auto copy(
199 Dest& dest, const Src& src, std::size_t width, std::size_t height, std::size_t depth) const -> void
200 {
201 cudaMemcpy3DParms parms;
202 auto uchar_width = width * sizeof(typename Src::element_type) / sizeof(unsigned char);
203 // using uchar_width instead of width because cudaMemcpy3D interprets the pointer's elements as
204 // unsigned char
205 parms.srcPtr = make_cudaPitchedPtr(
206 reinterpret_cast<unsigned char*>(src.get()), src.pitch(), uchar_width, height);
207 parms.dstPtr = make_cudaPitchedPtr(
208 reinterpret_cast<unsigned char*>(dest.get()), dest.pitch(), uchar_width, height);
209 parms.extent = make_cudaExtent(uchar_width, height, depth);
210 parms.kind = detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value;
211
212 CHECK(cudaMemcpy3D(&parms));
213 }
214};

236 /* 2D copies */

239 {
240 CHECK(cudaMemcpy2DAsync(dest.get(), dest.pitch(), src.get(), src.pitch(),
241 width * sizeof(typename Src::element_type), height,
242 detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value, stream_));
243 }
244
245 /* 3D copies */
246 template <class Dest, class Src>
247 inline auto copy(
248 Dest& dest, const Src& src, std::size_t width, std::size_t height, std::size_t depth) const -> void
249 {
250 cudaMemcpy3DParms parms;
251 auto uchar_width = width * sizeof(typename Src::element_type) / sizeof(unsigned char);
252 parms.srcPtr = make_cudaPitchedPtr(
253 reinterpret_cast<unsigned char*>(src.get()), src.pitch(), uchar_width, height);
254 parms.dstPtr = make_cudaPitchedPtr(
255 reinterpret_cast<unsigned char*>(dest.get()), dest.pitch(), uchar_width, height);
256 parms.extent = make_cudaExtent(uchar_width, height, depth);
257 parms.kind = detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value;
258
259 CHECK(cudaMemcpy3DAsync(&parms, stream_));
260 }
261
262 private:
263 cudaStream_t stream_;
264};

279template <class T, class CopyPolicy = sync_copy_policy>
280auto make_device_ptr(std::size_t length) -> device_ptr<T, CopyPolicy>
281{
282 auto size = length * sizeof(T);
283 auto p = static_cast<T*>(nullptr);
284 CHECK(cudaMalloc(&p, size));
285 return device_ptr<T, CopyPolicy>(detail::unique_device_ptr<T>(p), size);
286}

288template <class T, class CopyPolicy = sync_copy_policy>
289auto make_host_ptr(std::size_t length) -> host_ptr<T, CopyPolicy>
290{
291 auto p = static_cast<T*>(nullptr);
292 auto size = length * sizeof(T);
293 CHECK(cudaMallocHost(&p, size));
294 return host_ptr<T, CopyPolicy>(detail::unique_host_ptr<T>(p), size);
295}

297template <class T, class CopyPolicy = sync_copy_policy>
298auto make_device_ptr(std::size_t width, std::size_t height)
299 -> pitched_device_ptr<T, CopyPolicy, std::false_type>
300{
301 auto p = static_cast<T*>(nullptr);
302 auto pitch = std::size_t{};
303 CHECK(cudaMallocPitch(&p, &pitch, width * sizeof(T), height));
304 return pitched_device_ptr<T, CopyPolicy, std::false_type>(
305 detail::unique_device_ptr<T>(p), pitch, width, height);
306}

308template <class T, class CopyPolicy = sync_copy_policy>
309auto make_host_ptr(std::size_t width, std::size_t height) -> pitched_host_ptr<T, CopyPolicy, std::false_type>
310{
311 auto p = static_cast<T*>(nullptr);
312 auto pitch = width * sizeof(T);
313 CHECK(cudaMallocHost(&p, pitch * height));
314 return pitched_host_ptr<T, CopyPolicy, std::false_type>(
315 detail::unique_host_ptr<T>(p), pitch, width, height);
316}
317
318template <class T, class CopyPolicy = sync_copy_policy>
319auto make_device_ptr(std::size_t width, std::size_t height, std::size_t depth)
320 -> pitched_device_ptr<T, CopyPolicy, std::true_type>
321{
322 auto extent = make_cudaExtent(width * sizeof(T), height, depth);
323 auto pitchedPtr = cudaPitchedPtr{};
324 CHECK(cudaMalloc3D(&pitchedPtr, extent));
325 // omitting pitchedPtr.xsize and pitchedPtr.ysize as those are identical to width and height
326 return pitched_device_ptr<T, CopyPolicy, std::true_type>(
327 detail::unique_device_ptr<T>(static_cast<T*>(pitchedPtr.ptr)), pitchedPtr.pitch, width, height,
328 depth);
329}

338 return pitched_host_ptr<T, CopyPolicy, std::true_type>(

344/* Functor for explicit copies */
345template <class CopyPolicy>
346struct copy_ftor : public CopyPolicy
347{
348 /* 1D */
349 template <class Dest, class Src>
350 inline auto operator()(Dest& dest, const Src& src) ->
351 typename std::enable_if<(!Dest::has_pitch && !Src::has_pitch), void>::type
352 {
353 CopyPolicy::copy(dest, src, src.size());
354 }
355
356 /* 2D */
357 template <class Dest, class Src>
358 inline auto operator()(Dest& dest, const Src& src) ->
359 typename std::enable_if<(Dest::has_pitch && Src::has_pitch) && (!Dest::is3DPtr && !Src::is3DPtr),
360 void>::type
361 {
362 CopyPolicy::copy(dest, src, src.width(), src.height());
363 }
364
365 /* 3D */
366 template <class Dest, class Src>
367 inline auto operator()(Dest& dest, const Src& src) ->
368 typename std::enable_if<(Dest::has_pitch && Src::has_pitch) && (Dest::is3DPtr && Src::is3DPtr),
369 void>::type
370 {
371 CopyPolicy::copy(dest, src, src.width(), src.height(), src.depth());
372 }
373};
374}

379template <class Dest, class Src>
