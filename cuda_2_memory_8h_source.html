175 class sync_copy_policy
176 {
177  protected:
178  ~sync_copy_policy() = default;

180  /* 1D copies*/
181  template <class Dest, class Src>
182  inline auto copy(Dest& dest, const Src& src, std::size_t size) const -> void
183  {
184  CHECK(cudaMemcpy(dest.get(), src.get(), size,
185  detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value));
186  }
187 
188  /* 2D copies */
189  template <class Dest, class Src>
190  inline auto copy(Dest& dest, const Src& src, std::size_t width, std::size_t height) const -> void
191  {
192  CHECK(cudaMemcpy2D(dest.get(), dest.pitch(), src.get(), src.pitch(),
193  width * sizeof(typename Src::element_type), height,
194  detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value));
195  }
196 
197  /* 3D copies */
198  template <class Dest, class Src>
199  inline auto copy(
200  Dest& dest, const Src& src, std::size_t width, std::size_t height, std::size_t depth) const -> void
201  {
202  cudaMemcpy3DParms parms;
203  auto uchar_width = width * sizeof(typename Src::element_type) / sizeof(unsigned char);
204  // using uchar_width instead of width because cudaMemcpy3D interprets the pointer's elements as
205  // unsigned char
206  parms.srcPtr = make_cudaPitchedPtr(
207  reinterpret_cast<unsigned char*>(src.get()), src.pitch(), uchar_width, height);
208  parms.dstPtr = make_cudaPitchedPtr(
209  reinterpret_cast<unsigned char*>(dest.get()), dest.pitch(), uchar_width, height);
210  parms.extent = make_cudaExtent(uchar_width, height, depth);
211  parms.kind = detail::Direction<Src::underlying_type::target, Dest::underlying_type::target>::value;
212 
213  CHECK(cudaMemcpy3D(&parms));
214  }
215 };

348 {

360  typename std::enable_if<(Dest::has_pitch && Src::has_pitch) && (!Dest::is3DPtr && !Src::is3DPtr),
361  void>::type
362  {
363  CopyPolicy::copy(dest, src, src.width(), src.height());
364  }
365

368  inline auto operator()(Dest& dest, const Src& src) ->
369  typename std::enable_if<(Dest::has_pitch && Src::has_pitch) && (Dest::is3DPtr && Src::is3DPtr),

371  {
372  CopyPolicy::copy(dest, src, src.width(), src.height(), src.depth());
373  }
