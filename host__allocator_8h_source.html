74 
75  template <class U, memory_layout uml>
76  host_allocator(const host_allocator<U, uml>&) noexcept
77  {
78  static_assert(std::is_same<T, U>::value && mem_layout == uml, "Attempting to copy incompatible host allocator");
79  }
80 
81  ~host_allocator() = default;
82 
83  auto allocate(size_type n) -> pointer
84  {
85  auto p = static_cast<pointer>(nullptr);
86  auto err = cudaMallocHost(reinterpret_cast<void**>(&p), n * sizeof(value_type));
87  if(err != cudaSuccess)
88  detail::throw_error(err);
89  return pointer{p};
90  }
91 
92  auto deallocate(pointer p, size_type = 0) noexcept -> void
93  {
94  auto err = cudaFreeHost(reinterpret_cast<void*>(p));
95  if(err != cudaSuccess)
96  std::exit(err);
97  }
98  };
