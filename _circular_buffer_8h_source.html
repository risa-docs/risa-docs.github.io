39 template <class Item>
40 void push_back(Item&& item)
41 {
42 if(count_ < size_)
43 count_ = index_ + 1u;
44 // making Image go out of scope(otherwise performance drop if buffer filled)
45 auto release = std::move(buffer_[index_]);
46 buffer_[index_] = std::move(item);
47 index_ = (index_ + 1) % size_;
48 }
49
50 Object at(std::size_t index)
51 {
52 auto ret = std::move(buffer_[index]);
53 return ret;
54 }
55
56 std::size_t count() { return count_; }
57
58 bool full() { return count_ >= size_; }
59
60 void clear()
61 {
62 count_ = 0u;
63 index_ = 0u;
64 }
65
