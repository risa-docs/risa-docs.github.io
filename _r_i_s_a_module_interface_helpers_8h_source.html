40template <class implementation_t, typename implementation_ctor_t, typename implementation_dtor_t>
41class DynamicallyLoadedProcessorStage
42{
43 public:
44 DynamicallyLoadedProcessorStage(implementation_ctor_t* ctor, implementation_dtor_t* dtor,
45 const std::string& configFile, const std::string parameterSet, const int numberOfOutputs,
46 const int numberOfInputs)
47 : implementation_dtor{dtor}
48 {
49 implementation = ctor(configFile, parameterSet, numberOfOutputs, numberOfInputs);
50 }
51
52 ~DynamicallyLoadedProcessorStage() { implementation_dtor(implementation); }
53
54 using input_type = typename implementation_t::input_type;
55 using output_type = typename implementation_t::output_type;
56
57 void update(glados::Subject* s) { implementation->update(s); }
58 void process(input_type&& img, int input_idx)
59 {
60 implementation->process(std::forward<input_type>(img), input_idx);
61 }
62 output_type wait(int output_idx) { return implementation->wait(output_idx); }
63
64 private:
65 implementation_t* implementation;
66 implementation_dtor_t* implementation_dtor;
67};

117template <typename fctn_signature>
118fctn_signature* dlsym_s(void* libHandle, const std::string fctnName)
119{
120 dlerror();
121 fctn_signature* get_fctn = (fctn_signature*)dlsym(libHandle, fctnName.c_str());
122 const char* dlsym_error = dlerror();
123 if(dlsym_error)
124 {
125 BOOST_LOG_TRIVIAL(fatal) << "RISAModuleInterface: Could not load function '" << fctnName << "'.";
126 throw std::runtime_error("Unable to load library function.");
127 }
128
129 return get_fctn;
130}
