46inline auto checkCudaError(cudaError_t err, const char* file, int line) -> void
47{
48 if(err != cudaSuccess)
49 {
50 throw std::runtime_error{"CUDA call failed at " + std::string(file) + ":" + std::to_string(line)
51 + ": " + std::string(cudaGetErrorString(err))};
52 }
53}

55inline auto getCufftErrorString(cufftResult result) -> std::string
56{
57 switch(result)
58 {
59 case CUFFT_SUCCESS:
60 return "The cuFFT operation was successful";
61 case CUFFT_INVALID_PLAN:
62 return "cuFFT was passed an invalid plan handle";
63 case CUFFT_ALLOC_FAILED:
64 return "cuFFT failed to allocate GPU or CPU memory";
65 case CUFFT_INVALID_TYPE:
66 return "Invalid type";
67 case CUFFT_INVALID_VALUE:
68 return "Invalid pointer or parameter";
69 case CUFFT_INTERNAL_ERROR:
70 return "Driver or internal cuFFT library error";
71 case CUFFT_EXEC_FAILED:
72 return "Failed to execute an FFT on the GPU";
73 case CUFFT_SETUP_FAILED:
74 return "The cuFFT library failed to initialize";
75 case CUFFT_INVALID_SIZE:
76 return "User specified an invalid transform size";
77 case CUFFT_UNALIGNED_DATA:
78 return "Unaligned data";
79 case CUFFT_INCOMPLETE_PARAMETER_LIST:
80 return "Missing parameters in call";
81 case CUFFT_INVALID_DEVICE:
82 return "Execution of plan was on different GPU than plan creation";
83 case CUFFT_PARSE_ERROR:
84 return "Internal plan database error";
85 case CUFFT_NO_WORKSPACE:
86 return "No workspace has been provided prior to plan execution";
87 case CUFFT_NOT_IMPLEMENTED:
88 return "This feature was not implemented for your cuFFT version";
89 case CUFFT_LICENSE_ERROR:
90 return "NVIDIA license required. The file was either not found, is out of data, or otherwise invalid";
91 default:
92 return "Unknown error";
93 }
94}

105inline auto getCusparseErrorString(cusparseStatus_t stat) -> std::string

117 case CUSPARSE_STATUS_ARCH_MISMATCH:
118 return "The function requires a feature absent from the device architecture.";
119 case CUSPARSE_STATUS_MAPPING_ERROR:
120 return "An access to GPU memory space failed.";
121 case CUSPARSE_STATUS_EXECUTION_FAILED:
122 return "The GPU program failed to execute.";
123 case CUSPARSE_STATUS_INTERNAL_ERROR:
124 return "An internal cuSPARSE operation failed.";
125 case CUSPARSE_STATUS_MATRIX_TYPE_NOT_SUPPORTED:
126 return "The matrix type is not supported by this function.";
127 case CUSPARSE_STATUS_ZERO_PIVOT:

143inline auto getNvjpegErrorString(nvjpegStatus_t stat) -> std::string
144{
145 switch(stat)
146 {
147 case NVJPEG_STATUS_SUCCESS:
148 return "The operation completed successfully.";
149 case NVJPEG_STATUS_NOT_INITIALIZED:
150 return "The library handle was not initialized. A call to nvjpegCreate() is required to initialize "
151 "the handle.";
152 case NVJPEG_STATUS_INVALID_PARAMETER:
153 return "Wrong parameter was passed. For example, a null pointer as input data, or an image index not "
154 "in the allowed range.";
155 case NVJPEG_STATUS_BAD_JPEG:
156 return "Cannot parse the JPEG stream. Check that the encoded JPEG stream and its size parameters are "
157 "correct.";
158 case NVJPEG_STATUS_JPEG_NOT_SUPPORTED:
159 return "Attempting to decode a JPEG stream that is not supported by the nvJPEG library.";
160 case NVJPEG_STATUS_ALLOCATOR_FAILURE:
161 return "The user-provided allocator functions, for either memory allocation or for releasing the "
162 "memory, returned a non-zero code.";
163 case NVJPEG_STATUS_EXECUTION_FAILED:
164 return "Error during the execution of the device tasks.";
165 case NVJPEG_STATUS_ARCH_MISMATCH:
166 return "The device capabilities are not enough for the set of input parameters provided.";
167 case NVJPEG_STATUS_INTERNAL_ERROR:
168 return "Error during the execution of the device tasks.";
169 case NVJPEG_STATUS_IMPLEMENTATION_NOT_SUPPORTED:
170 return "Not supported.";
171 // case NVJPEG_STATUS_INCOMPLETE_BITSTREAM:
172 // return "Bitstream input data incomplete.";
173 default:
174 return "Unknown error";
175 }
