33template <class T, class CopyPolicy = sync_copy_policy>
34class HostMemoryManager : public CopyPolicy
35{
36 public:
37 using value_type = T;
38 using pointer_type_1D = host_ptr<T, CopyPolicy>;
39 using pointer_type_2D = pitched_host_ptr<T, CopyPolicy, std::false_type>;
40 using pointer_type_3D = pitched_host_ptr<T, CopyPolicy, std::true_type>;
41 using size_type = std::size_t;
42
43 protected:
44 inline auto make_ptr(size_type size) -> pointer_type_1D
45 {
46 return make_host_ptr<value_type, CopyPolicy>(size);
47 }
48
49 inline auto make_ptr(size_type width, size_type height) -> pointer_type_2D
50 {
51 return make_host_ptr<value_type, CopyPolicy>(width, height);
52 }
53
54 inline auto make_ptr(size_type width, size_type height, size_type depth) -> pointer_type_3D
55 {
56 return make_host_ptr<value_type, CopyPolicy>(width, height, depth);
57 }
58
59 template <typename Source>
60 inline auto copy(pointer_type_1D& dest, Source& src, size_type size) -> void
61 {
62 CopyPolicy::copy(dest, src, size);
63 }
64
65 template <typename Source>
66 inline auto copy(pointer_type_2D& dest, Source& src, size_type width, size_type height) -> void
67 {
68 CopyPolicy::copy(dest, src, width, height);
69 }
70
71 template <typename Source>
72 inline auto copy(pointer_type_3D& dest, Source& src, size_type width, size_type height, size_type depth)
73 -> void
74 {
75 CopyPolicy::copy(dest, src, width, height, depth);
76 }
77
78 ~HostMemoryManager() = default;
79};
