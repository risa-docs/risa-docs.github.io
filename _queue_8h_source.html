38template <class Object>
39class Queue
40{
41 public:
42 /*
43 * The default constructed Queue has limit 2, hence the member is 2.
44 */
45 Queue() : limit_{10u}, count_{0u} {}
46 explicit Queue(std::size_t limit) : limit_{limit}, count_{0u} {}
47
48 /*
49 * Item and Object are of the same type but we need this extra template to make use of the
50 * nice reference collapsing rules
51 */
52 template <class Item>
53 void push(Item&& item)
54 {
55 auto lock = std::unique_lock<decltype(mutex_)>{mutex_};
56 if(limit_ != 0u)
57 {
58 while(count_ >= limit_)
59 count_cv_.wait(lock);
60 }
61
62 queue_.push(std::forward<Item>(item));
63
64 if(limit_ != 0u)
65 ++count_;
66
67 item_cv_.notify_one();
68 }
69
70 template <class Item>
71 void pushNonBlocking(Item&& item)
72 {
73 auto lock = std::unique_lock<decltype(mutex_)>{mutex_};
74 if(limit_ != 0u)
75 {
76 while(count_ >= limit_)
77 {
78 queue_.pop();
79 --count_;
80 }
81 }
82
83 queue_.push(std::forward<Item>(item));
84
85 if(limit_ != 0u)
86 ++count_;
87
88 item_cv_.notify_one();
89 }
90
93 Object take()
94 {
95 auto lock = std::unique_lock<decltype(mutex_)>{mutex_};
96 while(queue_.empty())
97 item_cv_.wait(lock);
98
99 auto ret = std::move(queue_.front());
100 queue_.pop();
101
102 if(limit_ != 0u)
103 {
104 --count_;
105 count_cv_.notify_one();
106 }
107
108 return ret;
109 }
110
111 private:
112 const std::size_t limit_;
113 std::size_t count_;
114 mutable std::mutex mutex_;
115 std::condition_variable item_cv_, count_cv_;
116 std::queue<Object> queue_;
117};
118}
