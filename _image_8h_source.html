39template <class MemoryManager>
40class Image : public MemoryManager
41{
42 public:
43 using value_type = typename MemoryManager::value_type;
44 using pointer_type = typename MemoryManager::pointer_type_1D;
45 using size_type = typename MemoryManager::size_type;

47 private:

50 public:
51 Image() noexcept : data_{nullptr}, index_{0}, memoryPoolIndex_(0), plane_{0}, size_{0}, valid_{false} {}
52
53 ~Image()
54 {
55 if(valid_)
56 {
57 // if valid image goes out of scope, return it to MemoryPool -> automatic reuse
58 MemoryPool<MemoryManager>::instance()->returnMemory(std::move(*this));
59 }
60 }

62 Image(size_type size, size_type idx = 0, size_type planeID = 0, pointer_type img_data = nullptr)
63 : MemoryManager()
64 , data_{std::move(img_data)}
65 , index_{idx}
66 , memoryPoolIndex_(0)
67 , plane_(planeID)
68 , size_{size}
69 , valid_{true}
70 {
71 if(data_ == nullptr)
72 data_ = MemoryManager::make_ptr(size_);
73 }
74
75 Image(const Image& other)
76 : MemoryManager(other)
77 , index_{other.index_}
78 , memoryPoolIndex_(other.memoryPoolIndex_)
79 , plane_{other.plane_}
80 , size_{other.size_}
81 , valid_{other.valid_}
82 {
83 if(other.data_ == nullptr)
84 data_ = nullptr;
85 else
86 {
87 data_ = MemoryManager::make_ptr(size_);
88 MemoryManager::copy(data_, other.data_, size_);
89 }
90 }

105 auto invalid() -> void { valid_ = false; }

107 template <typename U>
108 auto operator=(const Image<U>& rhs) -> Image&
109 {
110 index_ = rhs.index();
111 plane_ = rhs.plane();
112 size_ = rhs.size();
113 start_ = rhs.start();
114 valid_ = rhs.valid();
115
116 if(rhs.container() == nullptr)
117 data_ = nullptr;
118 else
119 {
120 data_.reset(); // delete old content if any
121 data_ = MemoryManager::make_ptr(size_);
122 MemoryManager::copy(data_, rhs.container(), size_);
123 }
124
125 return *this;
126 }

142 auto operator=(Image&& rhs) noexcept -> Image&
143 {
144 data_ = std::move(rhs.data_);

156 }
