108  template <typename U>
109  auto operator=(const Image<U>& rhs) -> Image&
110  {
111  index_ = rhs.index();
112  plane_ = rhs.plane();

115  valid_ = rhs.valid();
116 
117  if(rhs.container() == nullptr)

119  else
