38 /* 1D specialization */
39 template <class T, class InternalAlloc>
40 class pool_allocator<T, memory_layout::pointer_1D, InternalAlloc>
41 {
42 public:
43 static constexpr auto mem_layout = InternalAlloc::mem_layout;
44 static constexpr auto mem_location = InternalAlloc::mem_location;
45 static constexpr auto alloc_needs_pitch = InternalAlloc::alloc_needs_pitch;
46
47 using value_type = T;
48 using pointer = typename InternalAlloc::pointer;
49 using const_pointer = typename InternalAlloc::const_pointer;
50 using size_type = typename InternalAlloc::size_type;
51 using difference_type = typename InternalAlloc::difference_type;
52 using propagate_on_container_copy_assignment = std::true_type;
53 using propagate_on_container_move_assignment = std::true_type;
54 using propagate_on_container_swap = std::true_type;
55 using is_always_equal = std::true_type;
56 using smart_pointer = typename InternalAlloc::template smart_pointer<std::function<void(T*)>>;
57
58 template <class U>
59 struct rebind
60 {
61 using other = pool_allocator<U, mem_layout, InternalAlloc>;
62 };
63
64 public:
65 pool_allocator() = default;
66
67 pool_allocator(size_type limit) noexcept
68 : alloc_{}, list_{}, n_{}, limit_{limit}, current_{0}
69 {}
70
71 pool_allocator(pool_allocator&& other) noexcept
72 : alloc_{std::move(other.alloc_)}, list_{std::move(other.list_)}
73 , n_{other.n_}, limit_{other.limit_}, current_{other.current_.load()}, moved_{other.moved_}
74 {
75 if(other.lock_.test_and_set())
76 lock_.test_and_set();
77 else
78 lock_.clear();
79
80 other.moved_ = true;
81 }

85 alloc_ = std::move(other.alloc_);
86 list_ = std::move(other.list_);

88 limit_ = other.limit_;
89 current_ = other.current_.load();
90 moved_ = other.moved_;
91
92 if(other.lock_.test_and_set())
93 lock_.test_and_set();
94 else
95 lock_.clear();
96
97 other.moved_ = true;

99 return *this;
100 }
101
102 pool_allocator(const pool_allocator& other) noexcept = delete;
103 auto operator=(const pool_allocator& other) noexcept -> pool_allocator& = delete;
104
105 ~pool_allocator()
106 {
107 // pool_allocator's contents have to be released manually
108 }
109
110 auto allocate(size_type n) -> pointer
111 {
112 if(moved_)
113 return pointer{nullptr};
114
115 if(n_ == 0)
116 n_ = n;
117
118 auto ret = static_cast<pointer>(nullptr);
119
120 if(limit_ != 0)
121 {
122 while(current_.load() >= limit_)
123 std::this_thread::yield();
124 }
125
126 while(lock_.test_and_set(std::memory_order_acquire))
127 std::this_thread::yield();
128
129 if(list_.empty())
130 ret = alloc_.allocate(n_);
131 else
132 {
133 ret = std::move(list_.front());
134 list_.pop_front();
135 }
136
137 ++current_;
138 lock_.clear(std::memory_order_release);
139
140 return ret;
141 }
142
143 auto allocate_smart(size_type n) -> smart_pointer
144 {
145 return smart_pointer{allocate(n), [this](T* ptr){ this->deallocate(ptr); }};
146 }
147
148 auto deallocate(pointer p, size_type = 0) noexcept -> void
149 {
150 if(moved_ || (p == nullptr))
151 return;
152
153 while(lock_.test_and_set(std::memory_order_acquire))
154 std::this_thread::yield();
155
156 list_.push_front(p);
157 --current_;
158 lock_.clear(std::memory_order_release);
159 }
160
161 auto release() noexcept -> void
162 {
163 if(moved_)
164 return;
165
166 while(lock_.test_and_set(std::memory_order_acquire))
167 std::this_thread::yield();
168
169 for(auto&& p : list_)
170 alloc_.deallocate(p, n_);
171
172 n_ = 0;
173 current_.store(0);
174
175 lock_.clear(std::memory_order_release);
176 }
177
178 private:
179 InternalAlloc alloc_;
180 std::atomic_flag lock_ = ATOMIC_FLAG_INIT;
181 std::forward_list<pointer> list_;
182 size_type n_;
183 size_type limit_;
184 std::atomic_size_t current_;
185 bool moved_ = false;
186 };
