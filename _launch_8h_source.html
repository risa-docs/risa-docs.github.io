36 inline auto roundUp(std::uint32_t num, std::uint32_t multiple) -> std::uint32_t
37 {
38  if(multiple == 0)
39  return num;
40 
41  if(num == 0)
42  return multiple;
43 
44  auto remainder = num % multiple;
45  if(remainder == 0)
46  return num;
47 
48  return num + multiple - remainder;
49 }
50 }

52 template <typename... Args>
53 auto launch(std::size_t input_size, void (*kernel)(Args...), Args... args) -> void
54 {
55  // calculate max potential blocks
56  auto block_size = int{};
57  auto min_grid_size = int{};
58  CHECK(cudaOccupancyMaxPotentialBlockSize(&min_grid_size, &block_size, kernel, 0, 0));
59 
60  // calculate de facto occupation based on input size
61  auto grid_size = (input_size + block_size - 1) / block_size;
62 
63  kernel<<<grid_size, block_size>>>(std::forward<Args>(args)...);
64  CHECK(cudaPeekAtLastError());
65 }

67 template <typename... Args>
68 auto launch(std::size_t input_width, std::size_t input_height, void (*kernel)(Args...), Args... args) -> void
69 {
70  auto threads = detail::roundUp(static_cast<unsigned int>(input_width * input_height), 1024u);
71  auto blocks = threads / 1024u;
72 
73  auto iwb = static_cast<unsigned int>(input_width) / blocks;
74  auto dim_x = ((iwb < 32u) && (iwb != 0u)) ? iwb : detail::roundUp(iwb, 32u);
75  auto ihb = static_cast<unsigned int>(input_height) / blocks;
76  auto dim_y = ((ihb < 32u) && (ihb != 0u)) ? ihb : detail::roundUp(ihb, 32u);
77  auto block_size = dim3{dim_x, dim_y};
78  auto grid_size = dim3{static_cast<unsigned int>((input_width + block_size.x - 1u) / block_size.x),
79  static_cast<unsigned int>((input_height + block_size.y - 1u) / block_size.y)};
