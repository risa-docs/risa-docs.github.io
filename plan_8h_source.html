106  template <class I, class O>
107  auto execute(I* idata, O* odata) -> void
108  {
109  static_assert(!std::is_same<I, O>::value, "Plan needs a direction for transformations between the same types.");
110  static_assert(detail::type_chooser<I, O>::value == transformation_type, "This plan can not be used for other types than originally specified.");
111  static_assert(std::is_same<O, typename detail::type_mapper<I>::type>::value, "Attempt to transform to an incompatible type.");
112 
113  cufft_exec(idata, odata);
114  }
115 
116  template <class I, class O>
117  auto execute(I* idata, O* odata, int direction) -> void
118  {
119  static_assert(std::is_same<I, O>::value, "Transformations between different types are implicitly inverse");
120  static_assert(detail::type_chooser<I, O>::value == transformation_type, "This plan can not be used for other types than originally specified.");
121  static_assert(std::is_same<O, typename detail::type_mapper<I>::type>::value, "Attempt to transform to an incompatible type.");
122 
123  cufft_exec(idata, odata, direction);
124  }

131  private:
132  auto handle_result(cufftResult res) const -> void
133  {
134  #pragma GCC diagnostic push
135  #pragma GCC diagnostic ignored "-Wswitch-enum"
136  switch(res)
137  {
138  case CUFFT_SUCCESS: break;
139  case CUFFT_INVALID_PLAN: throw invalid_argument{"The plan parameter is not a valid handle."};
140  case CUFFT_ALLOC_FAILED: throw bad_alloc{};
141  case CUFFT_INVALID_VALUE: throw invalid_argument{"One or more invalid parameters were passed to the API."};
142  case CUFFT_INTERNAL_ERROR: throw runtime_error{"An internal driver error was detected."};
143  case CUFFT_EXEC_FAILED: throw runtime_error{"cuFFT failed to execute the transform on the GPU."};
144  case CUFFT_SETUP_FAILED: throw runtime_error{"The cuFFT library failed to initialize."};
145  case CUFFT_INVALID_SIZE: throw invalid_argument{"One or more of the parameters is not a supported size."};
146  default: throw runtime_error{"Unknown error."};
147  }
148  #pragma GCC diagnostic pop
149  }
