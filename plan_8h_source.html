54 class plan
55 {
56 public:

58
59 plan() noexcept = default;
60 plan(int nx) : valid_{true} { handle_result(cufftPlan1d(&handle_, nx, transformation_type, 1)); }
61 plan(int nx, int ny) : valid_{true} { handle_result(cufftPlan2d(&handle_, nx, ny, transformation_type)); }
62 plan(int nx, int ny, int nz) : valid_{true} { handle_result(cufftPlan3d(&handle_, nx, ny, nz, transformation_type)); }

64 plan(int rank, int* n, int* inembed, int istride, int idist,
65 int* onembed, int ostride, int odist,

105 template <class I, class O>
106 auto execute(I* idata, O* odata) -> void
107 {
108 static_assert(!std::is_same<I, O>::value, "Plan needs a direction for transformations between the same types.");
109 static_assert(detail::type_chooser<I, O>::value == transformation_type, "This plan can not be used for other types than originally specified.");
110 static_assert(std::is_same<O, typename detail::type_mapper<I>::type>::value, "Attempt to transform to an incompatible type.");
111
112 cufft_exec(idata, odata);
113 }
114
115 template <class I, class O>
116 auto execute(I* idata, O* odata, int direction) -> void
117 {
118 static_assert(std::is_same<I, O>::value, "Transformations between different types are implicitly inverse");
119 static_assert(detail::type_chooser<I, O>::value == transformation_type, "This plan can not be used for other types than originally specified.");
120 static_assert(std::is_same<O, typename detail::type_mapper<I>::type>::value, "Attempt to transform to an incompatible type.");
121
122 cufft_exec(idata, odata, direction);
123 }

125 auto set_stream(cudaStream_t stream) -> void

130 private:
131 auto handle_result(cufftResult res) const -> void
132 {
133 #pragma GCC diagnostic push
134 #pragma GCC diagnostic ignored "-Wswitch-enum"
135 switch(res)
136 {
137 case CUFFT_SUCCESS: break;
138 case CUFFT_INVALID_PLAN: throw invalid_argument{"The plan parameter is not a valid handle."};
139 case CUFFT_ALLOC_FAILED: throw bad_alloc{};
140 case CUFFT_INVALID_VALUE: throw invalid_argument{"One or more invalid parameters were passed to the API."};
141 case CUFFT_INTERNAL_ERROR: throw runtime_error{"An internal driver error was detected."};
142 case CUFFT_EXEC_FAILED: throw runtime_error{"cuFFT failed to execute the transform on the GPU."};
143 case CUFFT_SETUP_FAILED: throw runtime_error{"The cuFFT library failed to initialize."};
144 case CUFFT_INVALID_SIZE: throw invalid_argument{"One or more of the parameters is not a supported size."};
145 default: throw runtime_error{"Unknown error."};
146 }
147 #pragma GCC diagnostic pop
148 }
