40template <class ImageSaver>
41class SinkStage
42 : public ImageSaver
43 , public InputSide<typename ImageSaver::input_type>
44 , public GenericStage
45{
46 public:
47 using input_type = typename ImageSaver::input_type;
48
49 public:
50 template <typename... Args>
51 SinkStage(const std::string& config, const std::string& parameterSet, Args&&... args)
52 : ImageSaver(std::forward<Args>(args)..., config, parameterSet), InputSide<input_type>()
53 {
54 this->input_queues_ = std::vector<glados::Queue<input_type>>(1);
55 this->identifier = parameterSet;
56 }
57
58 auto run() -> void
59 {
60 auto counter = 0;
61 while(true)
62 {
63 auto img = this->input_queues_[0].take();
64 if(img.valid())
65 {
66 ImageSaver::saveImage(std::move(img));
67 ++counter;
68 }
69 else
70 {
71 BOOST_LOG_TRIVIAL(info)
72 << "SinkStage<" << identifier << ">: Poisonous pill arrived, terminating.";
73 break; // poisonous pill
74 }
75 }
76 }
77
78 auto update(Subject* s) -> void { ImageSaver::update(s); }
79
80 void connectTo(std::shared_ptr<GenericStage> other_, int portIdx) override
81 {
82 BOOST_LOG_TRIVIAL(error) << "Cannot connect a sink stage to another stage. Requested to connect to "
83 << other_ << " at port " << portIdx;
84 return; // cannot connect saver to more stages
85 }
86
87 private:
88 std::string identifier;
89};
90}
