46 template <class D, class S>
47 auto create_3D_parms(D& d, const S& s, std::size_t x, std::size_t y, std::size_t z,
48 std::size_t d_off_x, std::size_t d_off_y, std::size_t d_off_z,
49 std::size_t s_off_x, std::size_t s_off_y, std::size_t s_off_z) -> cudaMemcpy3DParms
50 {
51 constexpr auto uchar_size = sizeof(unsigned char);
52 constexpr auto d_elem_size = sizeof(typename D::element_type) / uchar_size;
53 constexpr auto s_elem_size = sizeof(typename S::element_type) / uchar_size;
54
55 auto to_uchar_pos = [](std::size_t v, std::size_t size) { return v * size; };
56 auto x_ext = to_uchar_pos(x, d_elem_size);
57
58 auto extent = make_cudaExtent(x_ext, y, z);
59
60 constexpr auto d_size = sizeof(typename D::element_type);
61 constexpr auto s_size = sizeof(typename S::element_type);
62
63 auto d_pitch = d.pitch();
64 if(D::mem_location == memory_location::host)
65 d_pitch = x * d_size;
66
67 auto s_pitch = s.pitch();
68 if(S::mem_location == memory_location::host)
69 s_pitch = x * s_size;
70
71 auto d_pitched = make_cudaPitchedPtr(d.get(), d_pitch, x, y);
72 auto s_pitched = make_cudaPitchedPtr(s.get(), s_pitch, x, y);
73
74 auto d_pos_x = to_uchar_pos(d_off_x, d_elem_size);
75 auto d_pos_y = to_uchar_pos(d_off_y, d_elem_size);
76 auto d_pos_z = to_uchar_pos(d_off_z, d_elem_size);
77 auto s_pos_x = to_uchar_pos(s_off_x, s_elem_size);
78 auto s_pos_y = to_uchar_pos(s_off_y, s_elem_size);
79 auto s_pos_z = to_uchar_pos(s_off_z, s_elem_size);
80
81 auto d_pos = make_cudaPos(d_pos_x, d_pos_y, d_pos_z);
82 auto s_pos = make_cudaPos(s_pos_x, s_pos_y, s_pos_z);
83
84 auto parms = cudaMemcpy3DParms{};
85 parms.srcPos = s_pos;
86 parms.srcPtr = s_pitched;
87 parms.dstPos = d_pos;
88 parms.dstPtr = d_pitched;
89 parms.extent = extent;
90 parms.kind = detail::memcpy_direction<D::mem_location, S::mem_location>::value;
91
92 return parms;
93 }
94 }

96 class sync_policy
97 {
98 public:
99 template <class D, class S>
100 auto copy(D& d, const S& s, std::size_t x) const -> void
101 {
102 static_assert(!D::pitched_memory, "Destination memory must not be pitched for a 1D copy.");
103 static_assert(!S::pitched_memory, "Source memory must not be pitched for a 1D copy.");
104
105 constexpr auto size = sizeof(typename D::element_type);
106
107 auto err = cudaMemcpy(d.get(), s.get(), x * size, detail::memcpy_direction<D::mem_location, S::mem_location>::value);
108 if(err != cudaSuccess)
109 detail::throw_error(err);
110 }

112 template <class D, class S>
113 auto copy(D& d, const S& s, std::size_t x, std::size_t y) const -> void
114 {
115 static_assert((D::mem_location == memory_location::host) || D::pitched_memory, "Destination memory on the device must be pitched for a 2D copy.");
116 static_assert((S::mem_location == memory_location::host) || S::pitched_memory, "Source memory on the device must be pitched for a 2D copy.");
117
118 constexpr auto size = sizeof(typename D::element_type);
119
120 auto d_pitch = d.pitch();
121 if(D::mem_location == memory_location::host)
122 d_pitch = x * size;
123
124 auto s_pitch = s.pitch();
125 if(S::mem_location == memory_location::host)
126 s_pitch = x * size;
127
128 auto err = cudaMemcpy2D(d.get(), d_pitch, s.get(), s_pitch, x * size, y, detail::memcpy_direction<D::mem_location, S::mem_location>::value);
129 if(err != cudaSuccess)
130 detail::throw_error(err);
131 }

212 template <class D, class S>
213 auto copy(D& d, const S& s, cudaStream_t stream, std::size_t x) const -> void
214 {
215 static_assert(!D::pitched_memory, "Destination memory must not be pitched for a 1D copy");
216 static_assert(!S::pitched_memory, "Source memory must not be pitched for a 1D copy");
217 static_assert((D::mem_location == memory_location::device) || D::pinned_memory, "Destination on host memory must be pinned for asynchronous copies.");
218 static_assert((S::mem_location == memory_location::device) || S::pinned_memory, "Source on host memory must be pinned for asynchronous copies.");
219
220 constexpr auto size = sizeof(typename D::element_type);
221
222 auto err = cudaMemcpyAsync(d.get(), s.get(), x * size, detail::memcpy_direction<D::mem_location, S::mem_location>::value, stream);
223 if(err != cudaSuccess)
224 detail::throw_error(err);
225 }

229 {
230 static_assert((D::mem_location == memory_location::host) || D::pitched_memory, "Destination memory on the device must be pitched for a 2D copy.");
231 static_assert((S::mem_location == memory_location::host) || S::pitched_memory, "Source memory on the device must be pitched for a 2D copy.");
232 static_assert((D::mem_location == memory_location::device) || D::pinned_memory, "Destination memory on the host must be pinned for asynchronous copies.");
233 static_assert((S::mem_location == memory_location::device) || S::pinned_memory, "Source memory on the host must be pinned for asynchronous copies.");
234
235 constexpr auto size = sizeof(typename D::element_type);
236
237 auto d_pitch = d.pitch();
238 if(D::mem_location == memory_location::host)
239 d_pitch = x * size;

243 s_pitch = x * size;
244
245 auto err = cudaMemcpy2DAsync(d.get(), d_pitch, s.get(), s_pitch, x * size, y, detail::memcpy_direction<D::mem_location, S::mem_location>::value, stream);
246 if(err != cudaSuccess)
247 detail::throw_error(err);
248 }

281 template <class P>
282 auto fill(P& p, int value, std::size_t x) const
283 -> typename std::enable_if<P::mem_location == memory_location::host, void>::type
284 {
285 // in an ideal world we wouldn't spawn threads manually but use std::async instead.
286 // However, the world isn't ideal and std::async is completely useless when you want to launch "fire and forget" tasks.
287 auto f = [&]() { std::fill_n(p.get(), x, value); };
288 auto&& t = std::thread{f};
289 t.detach();
290 }
