47  template <class D, class S>
48  auto create_3D_parms(D& d, const S& s, std::size_t x, std::size_t y, std::size_t z,
49  std::size_t d_off_x, std::size_t d_off_y, std::size_t d_off_z,
50  std::size_t s_off_x, std::size_t s_off_y, std::size_t s_off_z) -> cudaMemcpy3DParms
51  {
52  constexpr auto uchar_size = sizeof(unsigned char);
53  constexpr auto d_elem_size = sizeof(typename D::element_type) / uchar_size;
54  constexpr auto s_elem_size = sizeof(typename S::element_type) / uchar_size;
55 
56  auto to_uchar_pos = [](std::size_t v, std::size_t size) { return v * size; };
57  auto x_ext = to_uchar_pos(x, d_elem_size);
58 
59  auto extent = make_cudaExtent(x_ext, y, z);
60 
61  constexpr auto d_size = sizeof(typename D::element_type);
62  constexpr auto s_size = sizeof(typename S::element_type);
63 
64  auto d_pitch = d.pitch();
65  if(D::mem_location == memory_location::host)
66  d_pitch = x * d_size;
67 
68  auto s_pitch = s.pitch();
69  if(S::mem_location == memory_location::host)
70  s_pitch = x * s_size;
71 
72  auto d_pitched = make_cudaPitchedPtr(d.get(), d_pitch, x, y);
73  auto s_pitched = make_cudaPitchedPtr(s.get(), s_pitch, x, y);
74 
75  auto d_pos_x = to_uchar_pos(d_off_x, d_elem_size);
76  auto d_pos_y = to_uchar_pos(d_off_y, d_elem_size);
77  auto d_pos_z = to_uchar_pos(d_off_z, d_elem_size);
78  auto s_pos_x = to_uchar_pos(s_off_x, s_elem_size);
79  auto s_pos_y = to_uchar_pos(s_off_y, s_elem_size);
80  auto s_pos_z = to_uchar_pos(s_off_z, s_elem_size);
81 
82  auto d_pos = make_cudaPos(d_pos_x, d_pos_y, d_pos_z);
83  auto s_pos = make_cudaPos(s_pos_x, s_pos_y, s_pos_z);
84 
85  auto parms = cudaMemcpy3DParms{};
86  parms.srcPos = s_pos;
87  parms.srcPtr = s_pitched;
88  parms.dstPos = d_pos;
89  parms.dstPtr = d_pitched;
90  parms.extent = extent;
91  parms.kind = detail::memcpy_direction<D::mem_location, S::mem_location>::value;
92 
93  return parms;
94  }
95  }

97  class sync_policy
98  {
99  public:
100  template <class D, class S>
101  auto copy(D& d, const S& s, std::size_t x) const -> void
102  {
103  static_assert(!D::pitched_memory, "Destination memory must not be pitched for a 1D copy.");
104  static_assert(!S::pitched_memory, "Source memory must not be pitched for a 1D copy.");
105 
106  constexpr auto size = sizeof(typename D::element_type);
107 
108  auto err = cudaMemcpy(d.get(), s.get(), x * size, detail::memcpy_direction<D::mem_location, S::mem_location>::value);
109  if(err != cudaSuccess)
110  detail::throw_error(err);
111  }

115  {

118 
119  constexpr auto size = sizeof(typename D::element_type);
120 
121  auto d_pitch = d.pitch();

123  d_pitch = x * size;

125  auto s_pitch = s.pitch();
126  if(S::mem_location == memory_location::host)

128

282  template <class P>
283  auto fill(P& p, int value, std::size_t x) const
284  -> typename std::enable_if<P::mem_location == memory_location::host, void>::type
285  {
286  // in an ideal world we wouldn't spawn threads manually but use std::async instead.
287  // However, the world isn't ideal and std::async is completely useless when you want to launch "fire and forget" tasks.
288  auto f = [&]() { std::fill_n(p.get(), x, value); };
289  auto&& t = std::thread{f};
290  t.detach();
291  }
