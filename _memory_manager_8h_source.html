36template <class T>
37class MemoryManager
38{
39 public:
40 using value_type = T;
41 using size_type = std::size_t;
42 using pointer_type_1D = ptr<T>;
43 using pointer_type_2D = pitched_ptr<T, std::false_type>;
44 using pointer_type_3D = pitched_ptr<T, std::true_type>;
45
46 public:
47 inline auto make_ptr(size_type size) -> pointer_type_1D
48 {
49 auto ptr = std::unique_ptr<T[]>(new value_type[size]);
50 return ptr<T, std::false_type>(std::move(ptr), size * sizeof(T));
51 }
52
53 inline auto make_ptr(size_type width, size_type height) -> pointer_type_2D
54 {
55 auto ptr = std::unique_ptr<T[]>(new value_type[width * height]);
56 return pitched_ptr<T, std::false_type>(std::move(ptr), width * sizeof(T), width, height);
57 }
58
59 inline auto make_ptr(size_type width, size_type height, size_type depth) -> pointer_type_3D
60 {
61 auto ptr = std::unique_ptr<T>(new value_type[width * height * depth]);
62 return pitched_ptr<T, std::true_type>(std::move(ptr), width * sizeof(T), width, height, depth);
63 }
64
65 inline auto copy(pointer_type_1D& dest, const pointer_type_1D& src, size_type size) -> void
66 {
67 std::copy(src.get(), src.get() + size, dest.get());
68 }
69
70 inline auto copy(pointer_type_2D& dest, const pointer_type_2D& src, size_type width, size_type height)
71 -> void
72 {
73 std::copy(src.get(), src.get() + width * height, dest.get());
74 }
75
76 inline auto copy(pointer_type_3D& dest, const pointer_type_3D& src, size_type width, size_type height,
77 size_type depth) -> void
78 {
79 std::copy(src.get(), src.get() + width * height * depth, dest.get());
80 }
81};
