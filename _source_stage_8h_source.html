44template <class ImageLoader>
45class SourceStage
46 : public ImageLoader
47 , public OutputSide<typename ImageLoader::output_type>
48 , public GenericStage
49{
50 public:
51 using output_type = typename ImageLoader::output_type;
52
53 public:
54 template <typename... Args>
55 SourceStage(const std::string& configPath, const std::string& parameterSet, Args&&... args)
56 : ImageLoader(std::forward<Args>(args)..., configPath, parameterSet)
57 , OutputSide<output_type>()
58 , done_{false}
59 , num_{0u}
60 , path_{configPath}
61 , parameterSet_{parameterSet}
62 {
63 }
64
65 auto run() -> void
66 {
67 while(true)
68 {
69 auto img = ImageLoader::loadImage();
70 if(!img.valid())
71 {
72 // all images loaded, send poisonous pill
73 BOOST_LOG_TRIVIAL(info) << "SourceStage: All source files '" << parameterSet_
74 << "' loaded, sending poisonous pill";
75 this->output(output_type());
76 done_ = true;
77 break;
78 }
79 this->output(std::move(img));
80 ++num_;
81 }
82 }
83
84 auto update(Subject* s) -> void { ImageLoader::update(s); }
85
86 void connectTo(std::shared_ptr<GenericStage> other_, int portIdx) override
87 {
88 using port_type = output_type;
89
90 auto port = std::unique_ptr<Port<port_type>>(new Port<port_type>);
91 auto other = std::dynamic_pointer_cast<InputSide<output_type>>(other_);
92 if(other == nullptr)
93 {
94 throw std::runtime_error("input/output type mismatch.");
95 }
96 port->attach(other, portIdx);
97 this->attach(std::move(port));
98 }
99
100 private:
101 bool done_;
102 std::uint32_t num_;
103 std::string path_;
104 std::string parameterSet_;
105 std::mutex m_;
106 std::condition_variable cv_;
107};
108}
