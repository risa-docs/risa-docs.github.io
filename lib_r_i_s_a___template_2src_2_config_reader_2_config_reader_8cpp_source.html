43bool ConfigReader::parseFile(const std::string& configPath)
44{
45 try
46 {
47 boost::property_tree::read_json(configPath.c_str(), cfg);
48 }
49 catch(boost::property_tree::ptree_bad_path const&)
50 {
51 std::cerr << "I/O error while reading file." << std::endl;
52 exit(1);
53 }
54 catch(boost::property_tree::json_parser_error const& je)
55 {
56 std::cerr << "Error parsing: " << je.filename() << " on line: " << je.line() << std::endl;
57 std::cerr << je.message() << std::endl;
58 exit(1);
59 }
60
61 isInitialized_ = true;
62
63 return true;
64}
