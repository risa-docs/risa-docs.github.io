34class copy_policy
35{
36 protected:
37 ~copy_policy() = default;
38
39 /* 1D copies */
40 template <class Dest, class Src>
41 inline auto copy(Dest& dest, const Src& src, std::size_t size) -> void
42 {
43 std::copy(src.get(), src.get() + size, dest.get());
44 }
45
46 /* 2D copies */
47 template <class Dest, class Src>
48 inline auto copy(Dest& dest, const Src& src, std::size_t width, std::size_t height) -> void
49 {
50 std::copy(src.get(), src.get() + width * height, dest.get());
51 }
52
53 /* 3D copies */
54 template <class Dest, class Src>
55 inline auto copy(Dest& dest, const Src& src, std::size_t width, std::size_t height, std::size_t depth)
56 -> void
57 {
58 std::copy(src.get(), src.get() + width * height * depth, dest.get());
59 }
60};
