40Template_Processor::Template_Processor(const std::string& configFile, const std::string& parameterSet,
41 const int numberOfOutputs, const int numberOfInputs)
42{
43
44 if(!readConfig(configFile))
45 {
46 throw std::runtime_error(
47 "risa::cuda::Template: Configuration file could not be loaded successfully.");
48 }
49
50 CHECK(cudaGetDeviceCount(&numberOfDevices_));
51
52 // when MemoryPool is required, register here:
53 // memoryPoolIdx_ =
54 // glados::MemoryPool<hostManagerType>::instance()->registerStage(memPoolSize_,
55 // numberOfPixels_ * numberOfPixels_);
56
57 // custom streams are necessary, because profiling with nvprof not possible with
58 //-default-stream per-thread option
59 for(auto i = 0; i < numberOfDevices_; i++)
60 {
61 CHECK(cudaSetDevice(i));
62 cudaStream_t stream;
63 CHECK(cudaStreamCreateWithPriority(&stream, cudaStreamNonBlocking, 0));
64 streams_[i] = stream;
65 }
66
67 // initialize output streams
68 results_ = std::vector<glados::Queue<output_type>>(numberOfOutputs_);
69
70 // initialize worker threads
71 for(auto i = 0; i < numberOfDevices_; i++)
72 {
73 imgs_[i] = std::vector<glados::Queue<input_type>>(numberOfInputs_);
74 processorThreads_[i] = std::thread{&Template_Processor::processor, this, i};
75 }
76
77 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Running " << numberOfDevices_ << " thread(s).";
78}

91auto Template_Processor::process(input_type&& img, int inputIdx) -> void
92{
93 if(img.valid())
94 {
95 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: (Device " << img.device() << " | Input "
96 << inputIdx << "): Image " << img.index() << " arrived.";
97 imgs_[img.device()][inputIdx].push(std::move(img));
98 }
99 else if(inputIdx == 0)
100 {
101 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Received sentinel, finishing.";
102
103 // send sentinel to processor threads and wait 'til they're finished
104 for(auto i = 0; i < numberOfDevices_; i++)
105 {
106 imgs_[i][0].push(input_type());
107 }
108
109 for(auto i = 0; i < numberOfDevices_; i++)
110 {
111 processorThreads_[i].join();
112 }
113
114 // push sentinel to results for next stage
115 for(int i = 0; i < numberOfOutputs_; i++)
116 results_[i].push(output_type());
117 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Finished.";
118 }
119}

123auto Template_Processor::processor(const int deviceID) -> void
124{
125 CHECK(cudaSetDevice(deviceID));
126 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Running thread for device " << deviceID;
127 while(true)
128 {
129 auto img = imgs_[deviceID][0].take();
130 if(!img.valid())
131 {
132 break;
133 }
134
135 // if necessary, request memory from MemoryPool here
136 auto ret = glados::MemoryPool<manager_type>::instance()->requestMemory(memoryPoolIdx_);
137
138 //<-- do work here -->
139
140 // in case of a CUDA stage, synchronization needs to be done here
141 // CHECK(cudaStreamSynchronize(streams_[deviceID]));
142
143 // wait until work on device is finished
144 results_[0].push(std::move(ret));
145
146 BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: (Device " << deviceID << " | Output " << 0
147 << "): Image " << ret.index() << " processed.";
