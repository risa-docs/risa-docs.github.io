92 auto Template_Processor::process(input_type&& img, int inputIdx) -> void
93 {
94  if(img.valid())
95  {
96  BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: (Device " << img.device() << " | Input "
97  << inputIdx << "): Image " << img.index() << " arrived.";
98  imgs_[img.device()][inputIdx].push(std::move(img));
99  }
100  else if(inputIdx == 0)
101  {
102  BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Received sentinel, finishing.";
103 
104  // send sentinel to processor threads and wait 'til they're finished
105  for(auto i = 0; i < numberOfDevices_; i++)
106  {
107  imgs_[i][0].push(input_type());
108  }
109 
110  for(auto i = 0; i < numberOfDevices_; i++)
111  {
112  processorThreads_[i].join();
113  }
114 
115  // push sentinel to results for next stage
116  for(int i = 0; i < numberOfOutputs_; i++)
117  results_[i].push(output_type());
118  BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Finished.";
119  }
120 }

125 {
126  CHECK(cudaSetDevice(deviceID));
127  BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: Running thread for device " << deviceID;
128  while(true)
129  {
130  auto img = imgs_[deviceID][0].take();
131  if(!img.valid())
132  {
133  break;
134  }
135 
136  // if necessary, request memory from MemoryPool here

138 
139  //<-- do work here -->

141  // in case of a CUDA stage, synchronization needs to be done here
142  // CHECK(cudaStreamSynchronize(streams_[deviceID]));
143 
144  // wait until work on device is finished
145  results_[0].push(std::move(ret));
146 
147  BOOST_LOG_TRIVIAL(debug) << "risa::cuda::Template: (Device " << deviceID << " | Output " << 0
148  << "): Image " << ret.index() << " processed.";
